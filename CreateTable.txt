	#region check table exist	
            var entities = _dbcontext.Model.GetEntityTypes();
            var tableName = entities.FirstOrDefault().GetTableName();
            var dbConn = _dbcontext.Database.GetDbConnection();
            dbConn.Open();
            string[] restrictions = new string[4];
            //restrictions[0] = "DbName";
            //restrictions[1] = "dbo";
            restrictions[2] = tableName;
            //restrictions[3] = "BASE TABLE";
            var dt = dbConn.GetSchema("Tables", restrictions);
            int exist = dt.Rows.Count;
            dbConn.Close();
	#endregion
